using System;
using System.Collections.Generic;
using System.Diagnostics;
using System.Linq;
using System.Threading.Tasks;
using Microsoft.Framework.Caching.Memory;
using Microsoft.Framework.Logging;
using NewOrbit.Time.HarvestClient.HarvestTypes;
using Newtonsoft.Json;

namespace NewOrbit.Time.HarvestClient
{
    public class HarvestClient
    {
        private readonly string subdomain;
        private readonly string userName;
        private readonly string password;
        private readonly ILogger logger;
        private readonly IMemoryCache cache;
        private RawHarvestClient rawClient;

        public HarvestClient(string subdomain, string userName, string password, ILogger logger, IMemoryCache cache)
        {
            this.subdomain = subdomain;
            this.userName = userName;
            this.password = password;
            this.logger = logger;
            this.cache = cache;
            this.rawClient = new RawHarvestClient(logger, cache, subdomain, userName, password);
        }
}
}
